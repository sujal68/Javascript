5. Sum of digits of a number

        function sumOfDigits() {
            let num = document.getElementById('sumDigitInput').value;
            let sum = 0;
            let n = Number(num);
            while (n > 0) {
                sum += n % 10;
                n = Math.floor(n / 10);
            }
            document.getElementById('sumDigitOutput').innerHTML = "Sum of digits: " + sum;
        }

6. Reverse a number

        function reverseNumber() {
            let n = Number(document.getElementById('reverseInput').value);
            let rev = 0;
            while (n > 0) {
                rev = rev * 10 + n % 10;
                n = Math.floor(n / 10);
            }
            document.getElementById('reverseOutput').innerHTML = "Reversed Number: " + rev;
        }

7. Check Palindrome number

        function checkPalindrome() {
            let num = document.getElementById('palindromeInput').value;
            let reversed = num.split('').reverse().join('');
            let res = (num == reversed) ? "Yes, Palindrome" : "Not Palindrome";
            document.getElementById('palindromeOutput').innerHTML = res;
        }

8. Check Prime number

        function checkPrime() {
            let n = Number(document.getElementById('primeInput').value);
            if (n <= 1) {
                document.getElementById('primeOutput').innerHTML = "Not Prime";
                return;
            }
            let isPrime = true;
            for (let i = 2; i <= Math.sqrt(n); i++) {
                if (n % i == 0) {
                    isPrime = false;
                    break;
                }
            }
            document.getElementById('primeOutput').innerHTML = isPrime ? "Prime" : "Not Prime";
        }

9. Fibonacci Series

        function fibonacciSeries() {
            let n = Number(document.getElementById('fibInput').value);
            let a = 0, b = 1;
            let series = [];
            for (let i = 1; i <= n; i++) {
                series.push(a);
                let next = a + b;
                a = b;
                b = next;
            }
            document.getElementById('fibOutput').innerHTML = series.join(', ');
        }

10. Factorial of a number

        function factorialNumber() {
            let n = Number(document.getElementById('factInput').value);
            let fact = 1;
            for (let i = 2; i <= n; i++) {
                fact *= i;
            }
            document.getElementById('factOutput').innerHTML = "Factorial: " + fact;
        }

11. Check Armstrong number

        function checkArmstrong() {
            let num = Number(document.getElementById('armstrongInput').value);
            let sum = 0;
            let n = num;
            let digits = n.toString().length;
            while (n > 0) {
                sum += Math.pow(n % 10, digits);
                n = Math.floor(n / 10);
            }
            let res = (sum === num) ? "Yes, Armstrong" : "Not Armstrong";
            document.getElementById('armstrongOutput').innerHTML = res;
        }

12. Sum of first and last digits of a number

        function sumFirstLast() {
            let num = document.getElementById('firstLastInput').value;
            let first = Number(num[0]);
            let last = Number(num[num.length - 1]);
            document.getElementById('firstLastOutput').innerHTML = "Sum: " + (first + last);
        }

13. Sum of middle digits of a number

        function sumMiddle() {
            let num = document.getElementById('middleInput').value;
            if (num.length <= 2) {
                document.getElementById('middleOutput').innerHTML = "No middle digits";
                return;
            }
            let middleDigits = num.slice(1, -1);
            let sum = 0;
            for (let digit of middleDigits) sum += Number(digit);
            document.getElementById('middleOutput').innerHTML = "Sum of middle digits: " + sum;
        }

14. Print table of a number

        function printTable() {
            let n = Number(document.getElementById('tableInput').value);
            let result = [];
            for (let i = 1; i <= 10; i++) {
                result.push(`${n}*${i}=${n * i}`);
            }
            document.getElementById('tableOutput').innerHTML = result.join('
');
        }
