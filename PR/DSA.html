Q.4 WAP to remove duplicate elements in a given array

Logic: Use loops to check duplicates and store unique elements in a new array. Example (C++ style): 


#include <iostream>
using namespace std;

int main() {
    int arr[] = {1, 2, 2, 3, 4, 4};
    int n = 6;

    int temp[6]; // new array to store unique elements
    int k = 0;   // index for temp

    for (int i = 0; i < n; i++) {
        bool found = false;

        // check if arr[i] already exists in temp
        for (int j = 0; j < k; j++) {
            if (arr[i] == temp[j]) {
                found = true;
                break;
            }
        }

        // if not found, add to temp
        if (!found) {
            temp[k] = arr[i];
            k++;
        }
    }

    cout << "Array after removing duplicates: ";
    for (int i = 0; i < k; i++) {
        cout << temp[i] << " ";
    }

    return 0;
}
